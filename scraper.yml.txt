name: Transparent Automation Scraper

on:
  schedule:
    # Executa 3 vezes ao dia em intervalos uniformes (00:00, 08:00, 16:00 UTC)
    - cron: '0 0,8,16 * * *'
  workflow_dispatch: # Permite rodar manualmente clicando num botão no GitHub

jobs:
  scrape:
    runs-on: ubuntu-latest
    
    steps:
      - name: Checkout do código
        uses: actions/checkout@v4

      - name: Configurar Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Instalar dependências do Scraper
        working-directory: ./scraper
        run: npm install

      - name: Instalar Navegadores do Playwright
        working-directory: ./scraper
        run: npx playwright install --with-deps chromium

      - name: Executar Operário Digital
        working-directory: ./scraper
        env:
          # Estas variáveis virão dos "Secrets" do seu repositório GitHub
          FIREBASE_SERVICE_ACCOUNT: ${{ secrets.FIREBASE_SERVICE_ACCOUNT }}
          USER_UID: ${{ secrets.USER_UID }}
          TARGET_LOGIN: ${{ secrets.TARGET_LOGIN }}
          TARGET_PASSWORD: ${{ secrets.TARGET_PASSWORD }}
        run: npm start
